pdf of the Maxwell-Boltzmann distribution

Here is the R [ ] source code:

######
bitmap(file="Maxwell-Boltzmann_distributionPDF.png",type="png256",width=4,height=4,res=300,pointsize=12)
par(mar=c(3,3,1,1))
mb <- function(x,a) { sqrt(2/pi) * x^2 * exp(-x^2/(2*a^2))/a^3 }
x <- seq(0,20,len=1000)
plot(x,mb(x,1),type="l",lwd=3)
lines(x,mb(x,2),col="red",lwd=3)
lines(x,mb(x,5),col="blue",lwd=3)
legend(12,0.5,legend=c("a=1","a=2","a=5"),col=c("black","red","blue"),lwd=3,bty="n")
dev.off()
